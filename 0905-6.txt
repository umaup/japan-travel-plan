        window.toggleChecklistItem = function(event) {
            if (!db || !userId) {
                // If not connected to a database (for local use), toggle class directly
                const li = event.currentTarget;
                li.classList.toggle('completed');
                return;
            }
            const li = event.currentTarget;
            const isCompleted = !li.classList.contains('completed');
            updateChecklistInDB(li.id, isCompleted);
        }

        function setupChecklistListeners() {
            // For local-only usage, we don't need a real-time listener
            // We can just keep the state in memory, or not at all.
            // For this version, let's keep it simple and just toggle the class.
        }
        
        async function updateChecklistInDB(id, isCompleted) {
            // This function is not used in the local-only version, but kept for clarity
            // so we don't accidentally remove it later.
            const checklistDocRef = doc(db, `artifacts/${appId}/public/data/checklists`, 'tripTasks');
            try {
                await setDoc(checklistDocRef, { [id]: isCompleted }, { merge: true });
            } catch (error) {
                console.error("Error updating checklist:", error);
            }
        }

        // --- TAB & UI LOGIC ---
        window.openTab = function(evt, tabName) {
            let i, tabcontent, tablinks;
            tabcontent = document.getElementsByClassName("tab-content");
            for (i = 0; i < tabcontent.length; i++) {
                tabcontent[i].style.display = "none";
            }
            tablinks = document.getElementsByClassName("tab-link");
            for (i = 0; i < tablinks.length; i++) {
                tablinks[i].className = tablinks[i].className.replace(" active", "");
            }
            const tabToShow = document.getElementById(tabName);
            if (tabToShow) {
                tabToShow.style.display = "block";
            }
            if (evt) {
                evt.currentTarget.className += " active";
            } else {
                document.querySelector('.tab-link[onclick*="tab-overview"]').className += " active";
            }
        }

        function highlightCurrentDayTab() {
            const today = new Date();
            today.setHours(0, 0, 0, 0); 
            
            document.querySelectorAll('.tab-link[data-date]').forEach(tab => {
                const tabDate = new Date(tab.dataset.date);
                if (today.getTime() === tabDate.getTime()) {
                    tab.classList.add('active-day');
                }
            });
        }
